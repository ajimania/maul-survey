🛠️ DB 초기 설정 (Supabase SQL Editor에서 1회 실행)

CREATE TABLE IF NOT EXISTS gohyang_participants (
  id              uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name            text NOT NULL,
  hometown_name   text,
  lat             float,
  lng             float,
  story           text,
  apply_reason    text,
  host_question   text,
  photo1_data     text,
  photo1_caption  text,
  photo2_data     text,
  photo2_caption  text,
  photo3_data     text,
  photo3_caption  text,
  created_at      timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS gohyang_cards (
  id         uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  question   text NOT NULL,
  type       text DEFAULT 'optional',
  created_at timestamptz DEFAULT now()
);

ALTER TABLE gohyang_participants ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all" ON gohyang_participants FOR ALL TO anon USING (true) WITH CHECK (true);

ALTER TABLE gohyang_cards ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all" ON gohyang_cards FOR ALL TO anon USING (true) WITH CHECK (true);

CREATE TABLE IF NOT EXISTS gohyang_reactions (
  id         uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  type       text NOT NULL,
  content    text NOT NULL,
  created_at timestamptz DEFAULT now()
);

ALTER TABLE gohyang_reactions ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all" ON gohyang_reactions FOR ALL TO anon USING (true) WITH CHECK (true);

ALTER PUBLICATION supabase_realtime ADD TABLE gohyang_reactions;
